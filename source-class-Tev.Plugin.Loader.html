  8 /**
  9  * Simple plugin loading utility.
 10  *
 11  * Provides framework for loading plugin content by convention, as follows:
 12  *
 13  * - Custom post types are loaded from an array, defined in `config/post_types.php`
 14  *   in the plugin's directory. The array is a set of key value pairs, where
 15  *   the key is the post type identifier and the value is the config to pass
 16  *   to `register_post_type`
 17  *
 18  * - Custom field groups are loaded from an array, in `config/field_groups.php`
 19  *   in the plugin's directory. The array is a set of arrays, defining each
 20  *   field group config to pass to `register_field_group`
 21  *
 22  * - ACF JSON config is loaded from a directory, at `config/acf-json/`
 23  *   in the plugin's directory. See http://www.advancedcustomfields.com/resources/local-json/
 24  *   for more information on this
 25  *
 26  * - Action callbacks are loaded from an array, in `config/actions.php` in the
 27  *   plugin's directory. The array is a set of key-value pairs, of action names
 28  *   to closure or action provider classes
 29  *
 30  * - Shortcodes are loaded from an array, in `config/shortcodes.php` in the
 31  *   plugin's directory. The array is a set of key value pairs, where
 32  *   the key is the shortcode name and the value is the config to pass
 33  *   to `add_shortcode`
 34  *
 35  * Usage:
 36  *
 37  * ```php
 38  * // Add the following to your plugin's config file
 39  *
 40  * tev_app()->fetch('plugin_loader')->load(__DIR__);
 41  * ```
 42  *
 43  */

 81      * Load all plugin configuration.
 82      *
 83      * @param  string             $basePath Plugin path
 84      * @return \Tev\Plugin\Loader           This, for chaining
 85      */
 86     public function load($basePath)
 87     {
 88         $this->basePath = $basePath;
 89         $this->renderer = new Renderer($this->getViewsPath());
 90 
 91         return
 92             $this
 93                 ->loadPostTypes()
 94                 ->loadFieldGroups()
 95                 ->loadAcfJson()
 96                 ->loadActions()
 97                 ->loadShortCodes()
 98                 ->loadOptionScreens();
 99     }

101     /**
102      * Load custom post types from configuration files.
103      *
104      * @return \Tev\Plugin\Loader This, for chaining
105      */
106     protected function loadPostTypes()
107     {
108         if ($config = $this->loadConfigFile('post_types.php')) {
109             $callbacks = array();
110 
111             // Create one callback for each post type, and register in
112             // init action
113 
114             foreach ($config as $postTypeName => $args) {
115                 $callbacks[] = $cb = function () use ($postTypeName, $args) {
116                     register_post_type($postTypeName, $args);
117                 };
118 
119                 add_action('init', $cb, 0);
120             }
121 
122             // Flush URL caches for (you need to register custom post types
123             // first)
124 
125             register_activation_hook($this->getPluginFile(), function () use ($callbacks) {
126                 foreach ($callbacks as $cb) {
127                     $cb();
128                 }
129 
130                 flush_rewrite_rules();
131             });
132         }
133 
134         return $this;
135     }

153     /**
154      * Load actions from configuration providers.
155      *
156      * @return \Tev\Plugin\Loader This, for chaining
157      */
158     protected function loadActions()
159     {
160         if ($config = $this->loadConfigFile('actions.php')) {
161             $app      = $this->app;
162             $renderer = $this->renderer;
163 
164             foreach ($config as $actionName => $provider) {
165                 if (is_string($provider) && is_subclass_of($provider, 'Tev\Plugin\Action\AbstractProvider')) {
166                     $ap = new $provider($this->app, $this->renderer);
167 
168                     add_action($actionName, function () use ($ap) {
169                         return call_user_func_array(array($ap, 'action'), func_get_args());
170                     }, $ap->priority(), $ap->numArgs());
171                 } elseif ($provider instanceof Closure) {
172                     add_action($actionName, function () use ($provider)
173                     {
174                         return call_user_func($provider, func_get_args());
175                     });
176                 }
177             }
178         }
179 
180         return $this;
181     }

183     /**
184      * Load ACF JSON config if supplied.
185      *
186      * @return \Tev\Plugin\Loader This, for chaining
187      */
188     protected function loadAcfJson()
189     {
190         $config = $this->getConfigPath() . '/acf-json';
191 
192         if (file_exists($config)) {
193             add_filter('acf/settings/load_json', function ($paths) use ($config) {
194                 $paths[] = $config;
195                 return $paths;
196             });
197         }
198 
199         return $this;
200     }

202     /**
203      * Load shortcodes from configuration files.
204      *
205      * @return \Tev\Plugin\Loader This, for chaining
206      */
207     protected function loadShortCodes()
208     {
209         if ($config = $this->loadConfigFile('shortcodes.php')) {
210             $renderer = $this->renderer;
211             $app      = $this->app;
212 
213             foreach ($config as $shortcode => $provider) {
214                 add_shortcode($shortcode, function ($attrs, $content) use ($app, $provider, $renderer)
215                 {
216                     if (is_string($provider) && is_subclass_of($provider, 'Tev\Plugin\Shortcode\AbstractProvider')) {
217                         $sp = new $provider($app, $renderer);
218                         return $sp->shortcode($attrs, $content);
219                     } elseif ($provider instanceof Closure) {
220                         return $provider($attrs, $content, $renderer);
221                     }
222                 });
223             }
224         }
225 
226         return $this;
227     }

289      * @return array|null       Array config or null if not found

291     protected function loadConfigFile($file)
292     {
293         $path = $this->getConfigPath() . '/' . $file;
294 
295         if (file_exists($path)) {
296             $config = include $path;
297             return is_array($config) ? $config : null;
298         }
299 
300         return null;
