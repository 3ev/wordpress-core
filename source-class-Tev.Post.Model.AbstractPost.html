226     /**
227      * Get the post excerpt.
228      *
229      * Adapted from wp_trim_excerpt() in wp-includes/formatting.php.
230      *
231      * @param  string|null $content Optional. Base content to use for excerpt
232      *                              if post excerpt isn't defined. Defaults to
233      *                              post content
234      * @return string
235      */
236     public function getExcerpt($content = null)
237     {
238         $raw = $exc = $this->base->post_excerpt;
239 
240         if (!strlen($exc)) {
241             $exc = $content === null ? $this->getRawContent() : $content;
242             $exc = strip_shortcodes($exc);
243             $exc = apply_filters('the_content', $exc);
244             $exc = str_replace(']]>', ']]&gt;', $exc);
245             $exc = wp_trim_words(
246                 $exc,
247                 apply_filters('excerpt_length', 55),
248                 apply_filters('excerpt_more', ' ' . '[&hellip;]')
249             );
250         }
251 
252         return apply_filters('wp_trim_excerpt', $exc, $raw);
253     }
