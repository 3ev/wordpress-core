 33     /**
 34      * Generate an array of breadcrumbs, based on the current page.
 35      *
 36      * Each item in the array is a hash containing 'title' and 'url' (which
 37      * might be null).
 38      *
 39      * You can use this to generate nice breadcrumb lists in your page.
 40      *
 41      * @param  \Closure|\Closure[] Optional. Set of user-defined callbacks that can
 42      *                             return breadcrumbs. If any of the supplied callbacks
 43      *                             return data, then their data will be used rather than
 44      *                             the defaults supplied by this method. Note that
 45      *                             the initial 'Home' crumb is always included
 46      * @return array
 47      */
 48     public function breadcrumbs($callbacks = null)
 49     {
 50         $breadcrumbs = array();
 51         $blogPage    = $this->postFactory->create(get_page(get_option('page_for_posts')));
 52 
 53         // Home page is always present
 54 
 55         $breadcrumbs[] = array(
 56             'title' => 'Home',
 57             'url'   => !is_front_page() ? get_option('home') : null
 58         );
 59 
 60         // Check if we've got any custom user defined crumbs
 61 
 62         if ($callbacks !== null) {
 63             if (!is_array($callbacks)) {
 64                 $callbacks = array($callbacks);
 65             }
 66 
 67             foreach ($callbacks as $cb) {
 68                 if ($cb instanceof Closure) {
 69                     if ($res = $cb()) {
 70                         $breadcrumbs[] = $res;
 71                     }
 72                 }
 73             }
 74         }
 75 
 76         // If we haven't got any user crumbs, proceed with default logic
 77 
 78         if (count($breadcrumbs) === 1) {
 79 
 80             // Blog home page (when different to front page)
 81 
 82             if (is_home() && !is_front_page()) {
 83                 $breadcrumbs[] = array(
 84                     'title' => $blogPage->getTitle(),
 85                     'url'   => null
 86                 );
 87             }
 88 
 89             // Custom post type archive page, no filtering
 90 
 91             if (is_archive()) {
 92 
 93                 // Custom post type archive
 94 
 95                 if (is_post_type_archive()) {
 96                     if (!$this->isFilteredArchive()) {
 97                         $breadcrumbs[] = array(
 98                             'title' => post_type_archive_title('', false),
 99                             'url'   => null
100                         );
101                     } else {
102                         $breadcrumbs[] = array(
103                             'title' => post_type_archive_title('', false),
104                             'url'   => get_post_type_archive_link(get_post_type())
105                         );
106 
107                         $breadcrumbs[] = array(
108                             'title' => $this->archiveTitle(),
109                             'url'   => null
110                         );
111                     }
112                 }
113 
114                 // Normal post archive
115 
116                 else {
117                     if (!$this->isBlogPageHome()) {
118                         $breadcrumbs[] = array(
119                             'title' => $blogPage->getTitle(),
120                             'url'   => $blogPage->getUrl()
121                         );
122                     }
123 
124                     if ($this->isFilteredArchive()) {
125                         $breadcrumbs[] = array(
126                             'title' => $this->archiveTitle(),
127                             'url'   => null
128                         );
129                     }
130                 }
131             }
132 
133             // Search page
134 
135             if (is_search()) {
136                 $breadcrumbs[] = array(
137                     'title' => 'Search',
138                     'url'   => null
139                 );
140             }
141 
142             // 404 page
143 
144             if (is_404()) {
145                 $breadcrumbs[] = array(
146                     'title' => '404 Not Found',
147                     'url'   => null
148                 );
149             }
150 
151             // Single post
152 
153             if (is_single()) {
154                 if (is_singular('post')) {
155                     if (!$this->isBlogPageHome()) {
156                         $breadcrumbs[] = array(
157                             'title' => $blogPage->getTitle(),
158                             'url'   => $blogPage->getUrl()
159                         );
160                     }
161                 } else {
162                     $postType = get_post_type();
163 
164                     $breadcrumbs[] = array(
165                         'title' => get_post_type_object($postType)->labels->name,
166                         'url'   => get_post_type_archive_link($postType)
167                     );
168                 }
169 
170                 $breadcrumbs[] = array(
171                     'title' => get_the_title(),
172                     'url'   => null
173                 );
174             }
175 
176             // Single Page
177 
178             if (is_page()) {
179                 global $post;
180 
181                 $page = $post;
182 
183                 $pageTree = array(
184                     array(
185                         'title' => $page->post_title,
186                         'url'   => null
187                     )
188                 );
189 
190                 while ($parentId = $page->post_parent) {
191                     $page = get_post($parentId);
192 
193                     array_unshift($pageTree, array(
194                         'title' => $page->post_title,
195                         'url'   => get_permalink($parentId)
196                     ));
197                 }
198 
199                 foreach ($pageTree as $t) {
200                     $breadcrumbs[] = $t;
201                 }
202             }
203         }
204 
205         return $breadcrumbs;
206     }

208     /**
209      * Render a view partial.
210      *
211      * @param  string $file   Partial file name
212      * @param  array  $params Optional variables to make available to partial
213      * @return string         Rendered partial
214      */
215     public function partial($file, array $variables = array())
216     {
217         extract($variables);
218 
219         if (substr($file, -4) !== '.php') {
220             $file .= '.php';
221         }
222 
223         $view = '';
224 
225         if ($tmpl = locate_template($file)) {
226             ob_start();
227             include($tmpl);
228             $view = ob_get_contents();
229             ob_end_clean();
230         }
231 
232         return $view;
233     }

235     /**
236      * Get a nice title for an archive page.
237      *
238      * Will return nice titles for taxonomy or date archive pages.
239      *
240      * Optionally config this method with formatting:
241      *
242      * - default ('Archive'): Title to display if archive page is not date or taxonomy
243      * - prefix (''): Text to prefix your title with
244      * - suffix (''): Text to suffix your title with
245      * - year_format ('Y'): Date format for 'year' archive pages
246      * - month_format ('F Y'): Date format for 'month' archive pages
247      * - day_format ('js F Y'): Date format for 'day' archive pages
248      *
249      * @param  array  $config Optional formatting config, as described above
250      * @return string         Archive title
251      */
252     public function archiveTitle(array $config = array())
253     {
254         $defaults = array(
255             'default'      => 'Archive',
256             'prefix'       => '',
257             'suffix'       => '',
258             'year_format'  => 'Y',
259             'month_format' => 'F Y',
260             'day_format'   => 'jS F Y'
261         );
262 
263         $config = array_merge($defaults, $config);
264 
265         $title = $config['default'];
266 
267         if (is_category() || is_tag() || is_tax()) {
268             $title = single_term_title('', false);
269         } elseif (is_year()) {
270             $title = Carbon::create(
271                 get_query_var('year')
272             )->format($config['year_format']);
273         } elseif (is_month()) {
274             $title = Carbon::create(
275                 get_query_var('year') ?: null,
276                 get_query_var('monthnum')
277             )->format($config['month_format']);
278         } elseif (is_day()) {
279             $title = Carbon::create(
280                 get_query_var('year') ?: null,
281                 get_query_var('monthnum') ?: null,
282                 get_query_var('day')
283             )->format($config['day_format']);
284         }
285 
286         return $config['prefix'] . $title . $config['suffix'];
287     }
